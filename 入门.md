# 1. 创建 Conda 环境
conda env create -f environment.yml
conda activate rag
# 2. 配置 DeepSeek API Key
# 编辑 .env 文件，填入 DEEPSEEK_API_KEY
# 3. 放入文档到 data/ 目录
# 4. 入库
python main.py ingest data/
# 5. 问答
python main.py query -i            # 交互模式
python main.py query -q "问题"      # 单次查询
python main.py query -c -q "问题"   # 显示参考文档
# 6. 运行测试
pytest --cov=src tests/
